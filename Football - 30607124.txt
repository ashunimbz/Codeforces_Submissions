int main()
{
  string s  ;
  cin>>s ; 
  for(int i  = 0  ; i< s.length()  ; ++i)
  {
      int c = 1 ; 
     for(int j = i + 1 ;  j < s.length() ; ++j)
     {
         if(s[j] == s[i])
          c++ ; 
        else
          break ; 
     }
     if(c >= 7)
     {
       cout<<"YES"  ;
       return 0  ;
     }
  }
  cout<<"NO"  ; 
return  0 ;  
}
